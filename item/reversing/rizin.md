```bash
# 설치 (kali는 기본 설치)
sudo apt update && sudo apt install -y rizin

# 바이너리 분석 시작
rizin <파일명>

# 진입 후 기본 분석
[0x00001080]> aaa        # 전체 자동 분석 (매우 자주 씀)

# 함수 목록 보기
[0x00001080]> afl        # 함수 리스트 (주소, 크기, 이름)

# 특정 함수(예: main) 디스어셈블
[0x00001080]> pdf @ main # main 함수 디스어셈블

# 문자열 검색
[0x00001080]> iz         # 포함된 문자열 전체 보기

# 섹션 정보
[0x00001080]> iS         # 섹션 정보

# 심볼 정보
[0x00001080]> is         # 심볼 테이블

# 바이너리 내수정 (패치)
[0x00001080]> wx <hex> @ <addr>  # 해당 주소에 hex값 패치

# 종료
[0x00001080]> q

# 실전 예시
rizin Compiled-1688545393558.Compiled
[0x00001080]> aaa
[0x00001080]> afl
[0x00001080]> pdf @ main
[0x00001080]> iz

# 분석 팁
# - aaa, afl, pdf @ main, iz 이 4개만 알아도 OSCP/CTF 실전 충분
# - 16진수 → 문자열: python3 -c 'print(bytes.fromhex("4973676e69727453")[::-1].decode())'
# - 리틀엔디안 주의 (역순)
# - 함수명, 문자열, 조건분기, 입력/출력 흐름만 파악해도 80% 해결

# 참고: rizin은 radare2 계열, 명령어 거의 동일
```
